# SQL-ASSIGNMENT-01

## Q1 New Customers Acquired in June 2023

```sql
SELECT P.PARTY_ID,
       PER.FIRST_NAME,
       PER.LAST_NAME,
       CM.INFO_STRING AS EMAIL,
       TN_PHONE.CONTACT_NUMBER ,
       P.CREATED_DATE AS ENTRY_DATE
FROM PARTY P
         JOIN PARTY_ROLE PR ON P.PARTY_ID = PR.PARTY_ID
         JOIN PERSON PER ON P.PARTY_ID = PER.PARTY_ID
         JOIN PARTY_CONTACT_MECH PCM ON P.PARTY_ID = PCM.PARTY_ID
         JOIN CONTACT_MECH CM ON PCM.CONTACT_MECH_ID = CM.CONTACT_MECH_ID
         JOIN PARTY_CONTACT_MECH PCM_PHONE ON P.party_id = PCM_PHONE.party_id
         JOIN TELECOM_NUMBER TN_PHONE ON PCM_PHONE.contact_mech_id = TN_PHONE.contact_mech_id
WHERE PR.ROLE_TYPE_ID = 'CUSTOMER'
    AND YEAR(P.CREATED_DATE) = 2023
  AND MONTH(P.CREATED_DATE) = 6
  AND CM.CONTACT_MECH_TYPE_ID = 'EMAIL_ADDRESS';
```

## Q8 Payment Captured but Not Shipped

```sql
SELECT OH.ORDER_ID,
OH.STATUS_ID AS order_status,
OPP.STATUS_ID AS payment_status,
S.STATUS_ID AS shipment_status
FROM order_header OH
JOIN ORDER_PAYMENT_PREFERENCE OPP ON OH.order_id = OPP.ORDER_ID
LEFT JOIN shipment S ON OH.ORDER_ID = S.PRIMARY_ORDER_ID
WHERE OPP.STATUS_ID = 'PAYMENT_SETTLED'
AND (S.STATUS_ID != 'SHIPMENT_SHIPPED' or S.STATUS_ID IS NULL);
```

## Q9 Orders Completed Hourly

```sql
SELECT count(order_id) AS TOTAL_ORDERS,
HOUR(STATUS_DATETIME) AS HOUR
FROM order_status
WHERE STATUS_ID = 'ORDER_COMPLETED'
GROUP BY HOUR(STATUS_DATETIME) ORDER BY HOUR;
```

## Q10 BOPIS Orders Revenue (Last Year)

```sql
SELECT COUNT(DISTINCT OISG.ORDER_ID) AS TOTAL_ORDERS,
SUM(OI.UNIT_PRICE) AS TOTAL_REVENUE
FROM order_item OI
JOIN order_item_ship_group_assoc OISGA ON OI.ORDER_ID = OISGA.ORDER_ID
AND OISGA.ORDER_ITEM_SEQ_ID = OI.ORDER_ITEM_SEQ_ID
JOIN order_item_ship_group OISG ON OISGA.ORDER_ID = OISG.ORDER_ID
AND OISGA.SHIP_GROUP_SEQ_ID = OISG.SHIP_GROUP_SEQ_ID
WHERE OISG.SHIPMENT_METHOD_TYPE_ID = 'STOREPICKUP'
AND ((YEAR(OI.LAST_UPDATED_STAMP) = YEAR(CURRENT_TIMESTAMP) - 1)
AND OI.STATUS_ID = 'ITEM_COMPLETED');
```

## Q11 Canceled Orders (Last Month)

```sql
SELECT CHANGE_REASON AS CANCELLATION_REASON,
COUNT(ORDER_ID) AS TOTAL_ORDERS
FROM order_status
WHERE STATUS_ID = 'ORDER_CANCELLED'
AND YEAR(STATUS_DATETIME) = YEAR(CURRENT_TIMESTAMP) - INTERVAL 1  MONTH
AND MONTH(STATUS_DATETIME) = MONTH(CURRENT_TIMESTAMP) - INTERVAL 1 MONTH
GROUP BY CHANGE_REASON;
```

## Q5.1 Shipping Addresses for October 2023 Orders

```sql
SELECT OH.ORDER_ID,
ORDR.PARTY_ID,
P.first_name AS CUSTOMER_NAME,
PA.ADDRESS1 AS STREET_ADDRESS,
PA.CITY,
PA.STATE_PROVINCE_GEO_ID,
PA.POSTAL_CODE,

JOIN order_contact_mech OCM ON OH.order_id = OCM.order_id
AND OCM.CONTACT_MECH_PURPOSE_TYPE_ID = 'SHIPPING_LOCATION'
JOIN postal_address PA ON PA.CONTACT_MECH_ID = OCM.CONTACT_MECH_ID
JOIN ORDER_ROLE ORDR ON OH.order_id = ORDR.order_id
AND ORDR.ROLE_TYPE_ID = 'PLACING_CUSTOMER'
JOIN PERSON P ON P.PARTY_ID = ORDR.PARTY_ID
WHERE OH.ORDER_DATE > "2023-9-30"
AND OH.ORDER_DATE < "2023-11-1";
```

## Q5.2 Orders from New York

```sql
SELECT OH.order_id,
```

## Q5.3 Top-Selling Product in New York

```sql
SELECT OI.PRODUCT_ID,
P.INTERNAL_NAME,
PA.CITY,
SUM(OI.QUANTITY) AS TOTAL_QUANTITY_SOLD,
SUM(OI.QUANTITY * OI.UNIT_PRICE) AS REVENUE
FROM order_header OH
JOIN ORDER_ITEM OI ON OH.ORDER_ID = OI.ORDER_ID
JOIN PRODUCT P ON OI.PRODUCT_ID = P.PRODUCT_ID
JOIN ORDER_CONTACT_MECH OCM ON OI.ORDER_ID = OCM.ORDER_ID
AND OCM.CONTACT_MECH_PURPOSE_TYPE_ID ='SHIPPING_LOCATION'
JOIN postal_address PA ON OCM.CONTACT_MECH_ID = PA.CONTACT_MECH_ID
WHERE PA.STATE_PROVINCE_GEO_ID = 'NY'
GROUP BY OI.PRODUCT_ID,
P.INTERNAL_NAME,
PA.CITY,
PA.STATE_PROVINCE_GEO_ID ORDER BY TOTAL_QUANTITY_SOLD DESC;
```

## Q8.1 Lost and Damaged Inventory

```sql

I.PRODUCT_ID,
I.FACILITY_ID,
IIV.REASON_ENUM_ID,

"VAR_STOLEN" ,
```

## Q8.2 Low Stock or Out of Stock Items Report

```sql
SELECT II.PRODUCT_ID,
P.PRODUCT_NAME,
II.FACILITY_ID,
II.AVAILABLE_TO_PROMISE_TOTAL AS ATP,
II.QUANTITY_ON_HAND_TOTAL AS QOH,
CURRENT_TIMESTAMP() AS DATE_CHECKED,
PF.MINIMUM_STOCK - II.QUANTITY_ON_HAND_TOTAL AS REORDER_THRESHOLD
FROM inventory_item II
JOIN product P ON P.product_id = II.product_id
JOIN product_facility PF ON PF.PRODUCT_ID = II.PRODUCT_ID
AND PF.facility_id = II.FACILITY_ID
WHERE II.QUANTITY_ON_HAND_TOTAL = 0
OR (II.QUANTITY_ON_HAND_TOTAL < PF.MINIMUM_STOCK);
```

## Q8 List of Warehouse Pickers

```sql
SELECT PR.party_id,
P.first_name,
P.last_name,
PR.role_type_id,
FP.facility_id,
CASE
WHEN (FP.from_date <= CURRENT_TIMESTAMP()
AND (FP.thru_date IS NULL
OR FP.thru_date >= CURRENT_TIMESTAMP())) THEN 'ACTIVE'
ELSE 'INACTIVE'
END
FROM party_role PR
JOIN PERSON P ON PR.party_id = P.party_id
JOIN FACILITY_PARTY FP ON FP.party_id = P.party_id
WHERE PR.role_type_id = 'WAREHOUSE_PICKER';
```
